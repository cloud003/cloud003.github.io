Posted on 2017-05-07 00:00:00 in python • 1 min read 

#从任意一个列表中得到这个列表所有元素的子集
import numpy as np
def found(n, seq):
    result = []
    s = np.array(seq)
    for i in range(2**n):
        e = list(bin(i))[2:]
        e = np.array(e) == '1'
        result.append(list(s[n-len(e):][e]))
    return result

lst = [1, 8, 4, 3, 5, 2]

list(filter(lambda x: sum(x)==10, found(6, lst)))

[[3, 5, 2], [8, 2], [1, 4, 5], [1, 4, 3, 2]]
